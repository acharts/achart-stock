function cloneObject(object){
    var o = {};
    for (var i in object) {
        o[i] = object[i];
    }
    return o;
}
function cloneArray(array){
    var arr = [];
    for (var i = 0; i < array.length; i++){
        if (typeof array[i] !== 'object') {
            arr.push(array[i]);
        } else {
            arr.push(cloneObject(array[i]));
        }
    }
    return arr;
}
